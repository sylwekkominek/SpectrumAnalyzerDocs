59template<typename T>
60std::vector<T> DataCalculatorBase<T>::calculateWithMoving()
61{
62
63 if(dataQueue.size() >= numberOfSignalsForProcessing)
64 {
65 data = std::vector<T>(numberOfSamples,initValue);
66
67 for(uint32_t i =0;i<numberOfSignalsForProcessing;++i)
68 {
69 std::transform(data.begin(), data.end(), dataQueue[i].begin(), data.begin(), algorithm);
70 }
71
72 dataQueue.pop_front();
73 return data;
74 }
75 return {};
76}
