230GLT_API GLTtext* gltCreateText(void)
231{
232 GLTtext *text = (GLTtext*)calloc(1, sizeof(GLTtext));
233
234 _GLT_ASSERT(text);
235
236 if (!text)
237 return GLT_NULL;
238
239 glGenVertexArrays(1, &text->_vao);
240 glGenBuffers(1, &text->_vbo);
241
242 _GLT_ASSERT(text->_vao);
243 _GLT_ASSERT(text->_vbo);
244
245 if (!text->_vao || !text->_vbo)
246 {
247 gltDeleteText(text);
248 return GLT_NULL;
249 }
250
251 glBindVertexArray(text->_vao);
252
253 glBindBuffer(GL_ARRAY_BUFFER, text->_vbo);
254
255 glEnableVertexAttribArray(_GLT_TEXT2D_POSITION_LOCATION);
256 glVertexAttribPointer(_GLT_TEXT2D_POSITION_LOCATION, _GLT_TEXT2D_POSITION_SIZE, GL_FLOAT, GL_FALSE, (_GLT_TEXT2D_VERTEX_SIZE * sizeof(GLfloat)), (const void*)(_GLT_TEXT2D_POSITION_OFFSET * sizeof(GLfloat)));
257
258 glEnableVertexAttribArray(_GLT_TEXT2D_TEXCOORD_LOCATION);
259 glVertexAttribPointer(_GLT_TEXT2D_TEXCOORD_LOCATION, _GLT_TEXT2D_TEXCOORD_SIZE, GL_FLOAT, GL_FALSE, (_GLT_TEXT2D_VERTEX_SIZE * sizeof(GLfloat)), (const void*)(_GLT_TEXT2D_TEXCOORD_OFFSET * sizeof(GLfloat)));
260
261 glBindVertexArray(0);
262
263 return text;
264}

293{
294 if (!text)

298
299 if (string)
300 strLength = strlen(string);

302 if (strLength)

307 return GL_TRUE;
308

311 }
312
313 text->_text = (char*)malloc((strLength + 1) * sizeof(char));
314
315 if (text->_text)

317 memcpy(text->_text, string, (strLength + 1) * sizeof(char));
318
319 text->_textLength = strLength;
320 text->_dirty = GL_TRUE;
321
322 return GL_TRUE;
323 }
324 }
325 else
326 {
327 if (text->_text)
328 {
329 free(text->_text);
330 text->_text = GLT_NULL;
331 }
332 else
333 {
334 return GL_TRUE;
335 }
336
337 text->_textLength = 0;
338 text->_dirty = GL_TRUE;
339
340 return GL_TRUE;
341 }
342
343 return GL_FALSE;

354GLT_API void gltViewport(GLsizei width, GLsizei height)
355{
356 _GLT_ASSERT(width > 0);
357 _GLT_ASSERT(height > 0);
358
359 const GLfloat left = 0.0f;
360 const GLfloat right = (GLfloat)width;
361 const GLfloat bottom = (GLfloat)height;
362 const GLfloat top = 0.0f;
363 const GLfloat zNear = -1.0f;
364 const GLfloat zFar = 1.0f;
365
366 const GLfloat projection[16] = {
367 (2.0f / (right - left)), 0.0f, 0.0f, 0.0f,
368 0.0f, (2.0f / (top - bottom)), 0.0f, 0.0f,
369 0.0f, 0.0f, (-2.0f / (zFar - zNear)), 0.0f,
370
371 -((right + left) / (right - left)),
372 -((top + bottom) / (top - bottom)),
373 -((zFar + zNear) / (zFar - zNear)),
374 1.0f,
375 };
376
377 memcpy(_gltText2DProjectionMatrix, projection, 16 * sizeof(GLfloat));
378}

515GLT_API GLfloat gltGetTextWidth(const GLTtext *text, GLfloat scale)
516{
517 if (!text || !text->_text)
518 return 0.0f;
519
520 GLfloat maxWidth = 0.0f;
521 GLfloat width = 0.0f;
522
523 _GLTglyph glyph;
524
525 char c;
526 int i;
527 for (i = 0; i < text->_textLength; i++)
528 {
529 c = text->_text[i];
530
531 if ((c == '\n') || (c == '\r'))
532 {
533 if (width > maxWidth)
534 maxWidth = width;
535
536 width = 0.0f;
537
538 continue;
539 }
540
541 if (!gltIsCharacterSupported(c))
542 {
543#ifdef GLT_UNKNOWN_CHARACTER
544 c = GLT_UNKNOWN_CHARACTER;
545 if (!gltIsCharacterSupported(c))
546 continue;
547#else
548 continue;
549#endif
550 }
551
552 glyph = _gltFontGlyphs2[c - _gltFontGlyphMinChar];
553
554 width += (GLfloat)glyph.w;
555 }
556
557 if (width > maxWidth)
558 maxWidth = width;
559
560 return maxWidth * scale;
561}

656GLT_API void _gltMat4Mult(const GLfloat lhs[16], const GLfloat rhs[16], GLfloat result[16])
657{
658 int c, r, i;
659
660 for (c = 0; c < 4; c++)
661 {
662 for (r = 0; r < 4; r++)
663 {
664 result[_GLT_MAT4_INDEX(r, c)] = 0.0f;
665
666 for (i = 0; i < 4; i++)
667 result[_GLT_MAT4_INDEX(r, c)] += lhs[_GLT_MAT4_INDEX(r, i)] * rhs[_GLT_MAT4_INDEX(i, c)];
668 }
669 }
670}

672GLT_API void _gltUpdateBuffers(GLTtext *text)
673{
674 if (!text || !text->_dirty)
675 return;
676
677 if (text->_vertices)
678 {
679 text->vertexCount = 0;
680
681 free(text->_vertices);
682 text->_vertices = GLT_NULL;
683 }
684
685 if (!text->_text || !text->_textLength)
686 {
687 text->_dirty = GL_FALSE;
688 return;
689 }
690
691 const GLsizei countDrawable = gltCountDrawableCharacters(text->_text);
692
693 if (!countDrawable)
694 {
695 text->_dirty = GL_FALSE;
696 return;
697 }
698
699 const GLsizei vertexCount = countDrawable * 2 * 3; // 3 vertices in a triangle, 2 triangles in a quad
700
701 const GLsizei vertexSize = _GLT_TEXT2D_VERTEX_SIZE;
702 GLfloat *vertices = (GLfloat*)malloc(vertexCount * vertexSize * sizeof(GLfloat));
703
704 if (!vertices)
705 return;
706
707 GLsizei vertexElementIndex = 0;
708
709 GLfloat glyphX = 0.0f;
710 GLfloat glyphY = 0.0f;
711
712 GLfloat glyphWidth;
713 const GLfloat glyphHeight = (GLfloat)_gltFontGlyphHeight;
714
715 const GLfloat glyphAdvanceX = 0.0f;
716 const GLfloat glyphAdvanceY = 0.0f;
717
718 _GLTglyph glyph;
719
720 char c;
721 int i;
722 for (i = 0; i < text->_textLength; i++)
723 {
724 c = text->_text[i];
725
726 if (c == '\n')
727 {
728 glyphX = 0.0f;
729 glyphY += glyphHeight + glyphAdvanceY;
730
731 continue;
732 }
733 else if (c == '\r')
734 {
735 glyphX = 0.0f;
736
737 continue;
738 }
739
740 if (!gltIsCharacterSupported(c))
741 {
742#ifdef GLT_UNKNOWN_CHARACTER
743 c = GLT_UNKNOWN_CHARACTER;
744 if (!gltIsCharacterSupported(c))
745 continue;
746#else
747 continue;
748#endif
749 }
750
751 glyph = _gltFontGlyphs2[c - _gltFontGlyphMinChar];
752
753 glyphWidth = (GLfloat)glyph.w;
754
755 if (glyph.drawable)
756 {
757 vertices[vertexElementIndex++] = glyphX;
758 vertices[vertexElementIndex++] = glyphY;
759 vertices[vertexElementIndex++] = glyph.u1;
760 vertices[vertexElementIndex++] = glyph.v1;
761
762 vertices[vertexElementIndex++] = glyphX + glyphWidth;
763 vertices[vertexElementIndex++] = glyphY + glyphHeight;
764 vertices[vertexElementIndex++] = glyph.u2;
765 vertices[vertexElementIndex++] = glyph.v2;
766
767 vertices[vertexElementIndex++] = glyphX + glyphWidth;
768 vertices[vertexElementIndex++] = glyphY;
769 vertices[vertexElementIndex++] = glyph.u2;
770 vertices[vertexElementIndex++] = glyph.v1;
771
772 vertices[vertexElementIndex++] = glyphX;
773 vertices[vertexElementIndex++] = glyphY;
774 vertices[vertexElementIndex++] = glyph.u1;
775 vertices[vertexElementIndex++] = glyph.v1;
776
777 vertices[vertexElementIndex++] = glyphX;
778 vertices[vertexElementIndex++] = glyphY + glyphHeight;
779 vertices[vertexElementIndex++] = glyph.u1;
780 vertices[vertexElementIndex++] = glyph.v2;
781
